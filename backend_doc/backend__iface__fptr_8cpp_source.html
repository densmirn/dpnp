44 DPNPFuncData_t get_dpnp_function_ptr(DPNPFuncName func_name, DPNPFuncType first_type, DPNPFuncType second_type)
45 {
46  DPNPFuncType local_second_type = (second_type == DPNPFuncType::DPNP_FT_NONE) ? first_type : second_type;
47 
48  func_map_t::const_iterator func_it = func_map.find(func_name);
49  if (func_it == func_map.cend())
50  {
51  throw std::runtime_error("DPNP Error: Unsupported function call."); // TODO print Function ID
52  }
53 
54  const map_1p_t& type1_map = func_it->second;
55  map_1p_t::const_iterator type1_map_it = type1_map.find(first_type);
56  if (type1_map_it == type1_map.cend())
57  {
58  throw std::runtime_error(
59  "DPNP Error: Function ID with unsupported first parameter type."); // TODO print Function ID
60  }
61 
62  const map_2p_t& type2_map = type1_map_it->second;
63  map_2p_t::const_iterator type2_map_it = type2_map.find(local_second_type);
64  if (type2_map_it == type2_map.cend())
65  {
66  throw std::runtime_error(
67  "DPNP Error: Function ID with unsupported second parameter type."); // TODO print Function ID
68  }
69 
70  DPNPFuncData_t func_info = type2_map_it->second;
71 
72  return func_info;
73 }

75 void* get_backend_function_name(const char* func_name, const char* type_name)
76 {
78  const char* supported_func_name = "dpnp_dot";
79  const char* supported_type1_name = "double";
80  const char* supported_type2_name = "float";
81  const char* supported_type3_name = "long";
82  const char* supported_type4_name = "int";
83 
85  if (!strncmp(func_name, supported_func_name, strlen(supported_func_name)))
86  {
87  if (!strncmp(type_name, supported_type1_name, strlen(supported_type1_name)))
88  {
89  return reinterpret_cast<void*>(dpnp_dot_c<double, double, double>);
90  }
91  else if (!strncmp(type_name, supported_type2_name, strlen(supported_type2_name)))
92  {
93  return reinterpret_cast<void*>(dpnp_dot_c<float, float, float>);
94  }
95  else if (!strncmp(type_name, supported_type3_name, strlen(supported_type3_name)))
96  {
97  return reinterpret_cast<void*>(dpnp_dot_c<long, long, long>);
98  }
99  else if (!strncmp(type_name, supported_type4_name, strlen(supported_type4_name)))
100  {
101  return reinterpret_cast<void*>(dpnp_dot_c<int, int, int>);
102  }
103  }
104 
105  throw std::runtime_error("DPNP Error: Unsupported function call");
106 }
