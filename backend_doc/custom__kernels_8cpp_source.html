41 template <typename _DataType>
42 void dpnp_matmul_c(void* array1_in, void* array2_in, void* result1, size_t size_m, size_t size_n, size_t size_k)
43 {
44  cl::sycl::event event;
45  _DataType* array_1 = reinterpret_cast<_DataType*>(array1_in);
46  _DataType* array_2 = reinterpret_cast<_DataType*>(array2_in);
47  _DataType* result = reinterpret_cast<_DataType*>(result1);
48 
49  if (!size_m || !size_n || !size_k)
50  {
51  return;
52  }
53 
54  if constexpr (std::is_same<_DataType, double>::value || std::is_same<_DataType, float>::value)
55  {
56  // using std::max for these ldx variables is required by math library
57  const std::int64_t lda = std::max<size_t>(1UL, size_k); // First dimensions of array_1
58  const std::int64_t ldb = std::max<size_t>(1UL, size_n); // First dimensions of array_2
59  const std::int64_t ldc = std::max<size_t>(1UL, size_n); // Fast dimensions of result
60 
61  event = mkl_blas::gemm(DPNP_QUEUE,
62  oneapi::mkl::transpose::nontrans,
63  oneapi::mkl::transpose::nontrans,
64  size_n,
65  size_m,
66  size_k,
67  _DataType(1),
68  array_2,
69  ldb,
70  array_1,
71  lda,
72  _DataType(0),
73  result,
74  ldc);
75  }
76  else
77  {
78  // input1: M x K
79  // input2: K x N
80  // result: M x N
81  const size_t dim_m = size_m; // shape1.front(); // First dimensions of array1
82  const size_t dim_n = size_n; // shape2.back(); // Last dimensions of array2
83  const size_t dim_k = size_k; // shape1.back(); // First dimensions of array2
84 
85  cl::sycl::range<2> gws(dim_m, dim_n); // dimensions are: "i" and "j"
86 
87  auto kernel_parallel_for_func = [=](cl::sycl::id<2> global_id) {
88  size_t i = global_id[0]; //for (size_t i = 0; i < size; ++i)
89  {
90  size_t j = global_id[1]; //for (size_t j = 0; j < size; ++j)
91  {
92  _DataType acc = _DataType(0);
93  for (size_t k = 0; k < dim_k; ++k)
94  {
95  const size_t index_1 = i * dim_k + k;
96  const size_t index_2 = k * dim_n + j;
97  acc += array_1[index_1] * array_2[index_2];
98  }
99  const size_t index_result = i * dim_n + j;
100  result[index_result] = acc;
101  }
102  }
103  };
104 
105  auto kernel_func = [&](cl::sycl::handler& cgh) {
106  cgh.parallel_for<class dpnp_matmul_c_kernel<_DataType>>(gws, kernel_parallel_for_func);
107  };
108 
109  event = DPNP_QUEUE.submit(kernel_func);
110  }
111  event.wait();
112 }

117 template <typename _DataType_input1, typename _DataType_input2, typename _DataType_output>
118 void dpnp_dot_c(void* array1_in, void* array2_in, void* result1, size_t size)
119 {
120  cl::sycl::event event;
121  _DataType_input1* array_1 = reinterpret_cast<_DataType_input1*>(array1_in);
122  _DataType_input2* array_2 = reinterpret_cast<_DataType_input2*>(array2_in);
123  _DataType_output* result = reinterpret_cast<_DataType_output*>(result1);
124 
125  if (!size)
126  {
127  return;
128  }
129 
130  if constexpr ((std::is_same<_DataType_input1, double>::value || std::is_same<_DataType_input1, float>::value) &&
131  std::is_same<_DataType_input2, _DataType_input1>::value &&
132  std::is_same<_DataType_output, _DataType_input1>::value)
133  {
134  event = mkl_blas::dot(DPNP_QUEUE,
135  size,
136  array_1,
137  1, // array_1 stride
138  array_2,
139  1, // array_2 stride
140  result);
141  event.wait();
142  }
143  else
144  {
145  _DataType_output* local_mem =
146  reinterpret_cast<_DataType_output*>(dpnp_memory_alloc_c(size * sizeof(_DataType_output)));
147 
148  // what about reduction??
149  cl::sycl::range<1> gws(size);
150 
151  auto kernel_parallel_for_func = [=](cl::sycl::id<1> global_id) {
152  const size_t index = global_id[0];
153  local_mem[index] = array_1[index] * array_2[index];
154  };
155 
156  auto kernel_func = [&](cl::sycl::handler& cgh) {
157  cgh.parallel_for<class dpnp_dot_c_kernel<_DataType_input1, _DataType_input2, _DataType_output>>(
158  gws, kernel_parallel_for_func);
159  };
160 
161  event = DPNP_QUEUE.submit(kernel_func);
162 
163  event.wait();
164 
165  auto policy = oneapi::dpl::execution::make_device_policy<
166  class dpnp_dot_c_kernel<_DataType_input1, _DataType_input2, _DataType_output>>(DPNP_QUEUE);
167 
168  _DataType_output accumulator = 0;
169  accumulator =
170  std::reduce(policy, local_mem, local_mem + size, _DataType_output(0), std::plus<_DataType_output>());
171  policy.queue().wait();
172 
173  result[0] = accumulator;
174 
175  free(local_mem, DPNP_QUEUE);
176  }
177 }

179 template <typename _DataType, typename _ResultType>
180 void dpnp_eig_c(const void* array_in, void* result1, void* result2, size_t size)
181 {
182  // TODO this kernel works with square 2-D array only
183 
184  // Kernel Type for calculation is double type
185  // because interface requires float type but calculations are expected in double type
186 
187  if (!size)
188  {
189  return;
190  }
191 
192  cl::sycl::event event;
193 
194  const _DataType* array = reinterpret_cast<const _DataType*>(array_in);
195  _ResultType* result_val = reinterpret_cast<_ResultType*>(result1);
196  _ResultType* result_vec = reinterpret_cast<_ResultType*>(result2);
197 
198  double* result_val_kern = reinterpret_cast<double*>(dpnp_memory_alloc_c(size * sizeof(double)));
199  double* result_vec_kern = reinterpret_cast<double*>(dpnp_memory_alloc_c(size * size * sizeof(double)));
200 
201  // type conversion. Also, math library requires copy memory because override
202  for (size_t it = 0; it < (size * size); ++it)
203  {
204  result_vec_kern[it] = array[it];
205  }
206 
207  const std::int64_t lda = std::max<size_t>(1UL, size);
208 
209  const std::int64_t scratchpad_size = mkl_lapack::syevd_scratchpad_size<double>(
210  DPNP_QUEUE, oneapi::mkl::job::vec, oneapi::mkl::uplo::upper, size, lda);
211 
212  double* scratchpad = reinterpret_cast<double*>(dpnp_memory_alloc_c(scratchpad_size * sizeof(double)));
213 
214  event = mkl_lapack::syevd(DPNP_QUEUE, // queue
215  oneapi::mkl::job::vec, // jobz
216  oneapi::mkl::uplo::upper, // uplo
217  size, // The order of the matrix A (0 <= n)
218  result_vec_kern, // will be overwritten with eigenvectors
219  lda,
220  result_val_kern,
221  scratchpad,
222  scratchpad_size);
223  event.wait();
224 
225  dpnp_memory_free_c(scratchpad);
226 
227  for (size_t it1 = 0; it1 < size; ++it1)
228  {
229  result_val[it1] = result_val_kern[it1];
230  for (size_t it2 = 0; it2 < size; ++it2)
231  {
232  // copy + transpose
233  result_vec[it2 * size + it1] = result_vec_kern[it1 * size + it2];
234  }
235  }
236 
237  dpnp_memory_free_c(result_val_kern);
238  dpnp_memory_free_c(result_vec_kern);
239 }
