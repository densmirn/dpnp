160 class DPNPC_id final
161 {
162 public:
163  using value_type = _Tp;
164  using iterator = DPNP_USM_iterator<value_type>;

168 
170  DPNPC_id(void* __ptr, const std::vector<size_type>& __shape)
171  {
172  if (__ptr == nullptr)
173  {
174  return;
175  }
176 
177  data = reinterpret_cast<pointer>(__ptr);
178  size = 1; // means scalar at this stage
179  output_size = 1; // if input size is not zero it means we have scalar as output
180 
181  if (!__shape.empty())
182  {
183  shape_size = __shape.size();
184  shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(shape_size * sizeof(size_type)));
185  std::copy(__shape.begin(), __shape.end(), shape);
186 
187  shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(shape_size * sizeof(size_type)));
188  get_shape_offsets_inkernel<size_type>(shape, shape_size, shape_strides);
189 
190  size = std::accumulate(__shape.begin(), __shape.end(), size_type(1), std::multiplies<size_type>());
191  if (size == 0)
192  { // shape might be shape[3, 4, 0, 6]. This means no input memory and no output expected
193  output_size = 0;
194  }
195  }
196  }
197 
198  DPNPC_id() = delete;
199 
200  ~DPNPC_id()
201  {
202  dpnp_memory_free_c(shape);
203  dpnp_memory_free_c(shape_strides);
204  dpnp_memory_free_c(output_shape);
205  dpnp_memory_free_c(output_shape_strides);
206  dpnp_memory_free_c(sycl_output_xyz);
207  dpnp_memory_free_c(sycl_input_xyz);
208  }

218  inline void set_axis(size_type __axis)
219  {
220  if (__axis < shape_size)
221  {
222  axis = __axis;
223  axis_use = true;
224 
225  output_shape_size = shape_size - 1;
226  const size_type output_shape_size_in_bytes = output_shape_size * sizeof(size_type);
227 
228  output_shape = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
229  size_type* output_shape_it = std::copy(shape, shape + axis, output_shape);
230  std::copy(shape + axis + 1, shape + shape_size, output_shape_it);
231 
232  output_size = std::accumulate(
233  output_shape, output_shape + output_shape_size, size_type(1), std::multiplies<size_type>());
234 
235  output_shape_strides = reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_shape_size_in_bytes));
236  get_shape_offsets_inkernel<size_type>(output_shape, output_shape_size, output_shape_strides);
237 
238  // make thread private storage for each shape by multiplying memory
239  sycl_output_xyz =
240  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * output_shape_size_in_bytes));
241  sycl_input_xyz =
242  reinterpret_cast<size_type*>(dpnp_memory_alloc_c(output_size * shape_size * sizeof(size_type)));
243  }
244  // TODO exception if wrong axis? need common function for throwing exceptions
245  // TODO need conversion from negative axis to positive one
246  }

269 private:
271  size_type get_input_begin_offset(size_type output_global_id) const
272  {
273  size_type input_global_id = 0;
274  if (axis_use)
275  {
276  assert(output_global_id < output_size);
277 
278  // use thread private storage
279  size_type* sycl_output_xyz_thread = sycl_output_xyz + (output_global_id * output_shape_size);
280  size_type* sycl_input_xyz_thread = sycl_input_xyz + (output_global_id * shape_size);
281 
282  get_xyz_by_id_inkernel(output_global_id, output_shape_strides, output_shape_size, sycl_output_xyz_thread);
283 
284  for (size_t iit = 0, oit = 0; iit < shape_size; ++iit)
285  {
286  if (iit == axis)
287  {
288  sycl_input_xyz_thread[iit] = 0; // put begin point of the axis
289  }
290  else
291  {
292  sycl_input_xyz_thread[iit] = sycl_output_xyz_thread[oit];
293  ++oit;
294  }
295  }
296 
297  input_global_id = get_id_by_xyz_inkernel(sycl_input_xyz_thread, shape_size, shape_strides);
298  }
299 
300  return input_global_id;
301  }
302 
304  size_type get_input_end_length() const
305  {
306  if (!axis_use)
307  {
308  return size;
309  }
310 
311  const size_type dim_size = shape[axis];
312  const size_type dim_pitch = shape_strides[axis];
313 
314  return (dim_size * dim_pitch);
315  }

334  size_type* sycl_input_xyz = nullptr;
