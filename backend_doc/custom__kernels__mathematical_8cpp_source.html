38 template <typename _DataType>
39 void custom_elemwise_absolute_c(void* array1_in, void* result1, size_t size)
40 {
41  if (!size)
42  {
43  return;
44  }
45 
46  cl::sycl::event event;
47  _DataType* array1 = reinterpret_cast<_DataType*>(array1_in);
48  _DataType* result = reinterpret_cast<_DataType*>(result1);
49 
50  if constexpr (std::is_same<_DataType, double>::value || std::is_same<_DataType, float>::value)
51  {
52  // https://docs.oneapi.com/versions/latest/onemkl/abs.html
53  event = oneapi::mkl::vm::abs(DPNP_QUEUE, size, array1, result);
54  }
55  else
56  {
57  cl::sycl::range<1> gws(size);
58  auto kernel_parallel_for_func = [=](cl::sycl::id<1> global_id) {
59  const size_t idx = global_id[0];
60 
61  if (array1[idx] >= 0)
62  {
63  result[idx] = array1[idx];
64  }
65  else
66  {
67  result[idx] = -1 * array1[idx];
68  }
69  };
70 
71  auto kernel_func = [&](cl::sycl::handler& cgh) {
72  cgh.parallel_for<class custom_elemwise_absolute_c_kernel<_DataType>>(gws, kernel_parallel_for_func);
73  };
74 
75  event = DPNP_QUEUE.submit(kernel_func);
76  }
77 
78  event.wait();
79 }

90 void dpnp_floor_divide_c(void* array1_in, void* array2_in, void* result1, size_t size)
91 {
92  cl::sycl::event event;
93  _DataType_input1* array1 = reinterpret_cast<_DataType_input1*>(array1_in);
94  _DataType_input2* array2 = reinterpret_cast<_DataType_input2*>(array2_in);
95  _DataType_output* result = reinterpret_cast<_DataType_output*>(result1);
96 
97  if constexpr ((std::is_same<_DataType_input1, double>::value || std::is_same<_DataType_input1, float>::value) &&
98  std::is_same<_DataType_input2, _DataType_input1>::value)
99  {
100  event = oneapi::mkl::vm::div(DPNP_QUEUE, size, array1, array2, result);
101  event.wait();
102  event = oneapi::mkl::vm::floor(DPNP_QUEUE, size, result, result);
103  }
104  else
105  {
106  cl::sycl::range<1> gws(size);
107  auto kernel_parallel_for_func = [=](cl::sycl::id<1> global_id) {

109  {
110  _DataType_input1 input_elem1 = array1[i];
111  _DataType_input2 input_elem2 = array2[i];
112  double div = (double)input_elem1 / (double)input_elem2;
113  result[i] = static_cast<_DataType_output>(cl::sycl::floor(div));
114  }
115  };

169 void dpnp_remainder_c(void* array1_in, void* array2_in, void* result1, size_t size)
170 {
171  cl::sycl::event event;
172  _DataType_input1* array1 = reinterpret_cast<_DataType_input1*>(array1_in);
173  _DataType_input2* array2 = reinterpret_cast<_DataType_input2*>(array2_in);
174  _DataType_output* result = reinterpret_cast<_DataType_output*>(result1);
175 
176  if constexpr ((std::is_same<_DataType_input1, double>::value || std::is_same<_DataType_input1, float>::value) &&
177  std::is_same<_DataType_input2, _DataType_input1>::value)
178  {
179  event = oneapi::mkl::vm::fmod(DPNP_QUEUE, size, array1, array2, result);
180  event.wait();
181  event = oneapi::mkl::vm::add(DPNP_QUEUE, size, result, array2, result);
182  event.wait();
183  event = oneapi::mkl::vm::fmod(DPNP_QUEUE, size, result, array2, result);
184  }
185  else
186  {
187  cl::sycl::range<1> gws(size);
188  auto kernel_parallel_for_func = [=](cl::sycl::id<1> global_id) {
189  size_t i = global_id[0]; /*for (size_t i = 0; i < size; ++i)*/
190  {
191  _DataType_input1 input_elem1 = array1[i];
192  _DataType_input2 input_elem2 = array2[i];
193  double fmod = cl::sycl::fmod((double)input_elem1, (double)input_elem2);
194  double add = fmod + input_elem2;
195  result[i] = cl::sycl::fmod(add, (double)input_elem2);
196  }
197  };
198 
199  auto kernel_func = [&](cl::sycl::handler& cgh) {
200  cgh.parallel_for<class dpnp_remainder_c_kernel<_DataType_input1, _DataType_input2, _DataType_output>>(
201  gws, kernel_parallel_for_func);
202  };
203 
204  event = DPNP_QUEUE.submit(kernel_func);
205  }
206 
207  event.wait();
208 }
