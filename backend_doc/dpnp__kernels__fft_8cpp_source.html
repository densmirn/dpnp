38 template <typename _DataType_input, typename _DataType_output>
39 void dpnp_fft_fft_c(const void* array1_in,
40  void* result1,
41  const long* input_shape,
42  const long* output_shape,
43  size_t shape_size,
44  long axis)
45 {
46  const size_t result_size = std::accumulate(output_shape, output_shape + shape_size, 1, std::multiplies<size_t>());
47  if (!(result_size && shape_size))
48  {
49  return;
50  }
51 
52  cl::sycl::event event;
53 
54  const _DataType_input* array_1 = reinterpret_cast<const _DataType_input*>(array1_in);
55  _DataType_output* result = reinterpret_cast<_DataType_output*>(result1);
56 
57  // kernel specific temporal data
58  long* output_shape_offsets = reinterpret_cast<long*>(dpnp_memory_alloc_c(shape_size * sizeof(long)));
59  long* input_shape_offsets = reinterpret_cast<long*>(dpnp_memory_alloc_c(shape_size * sizeof(long)));
60  // must be a thread local storage.
61  long* xyz = reinterpret_cast<long*>(dpnp_memory_alloc_c(result_size * shape_size * sizeof(long)));
62  long* axis_iterator = reinterpret_cast<long*>(dpnp_memory_alloc_c(result_size * shape_size * sizeof(long)));
63 
64  get_shape_offsets_inkernel<long>(output_shape, shape_size, output_shape_offsets);
65  get_shape_offsets_inkernel<long>(input_shape, shape_size, input_shape_offsets);
66 
67  cl::sycl::range<1> gws(result_size);
68  auto kernel_parallel_for_func = [=](cl::sycl::id<1> global_id) {
69  size_t output_id = global_id[0];
70 
71  double sum_real = 0.0;
72  double sum_imag = 0.0;
73  // need to replace these arrays by thread local storage
74  long* xyz_thread = xyz + (output_id * shape_size);
75  long* axis_iterator_thread = axis_iterator + (output_id * shape_size);
76 
77  get_xyz_by_id_inkernel(output_id, output_shape_offsets, shape_size, xyz_thread);
78  for (size_t i = 0; i < shape_size; ++i)
79  {
80  axis_iterator_thread[i] = xyz_thread[i];
81  }
82 
83  const long axis_length = output_shape[axis];
84  for (long it = 0; it < axis_length; ++it)
85  {
86  double in_real = 0.0;
87  double in_imag = 0.0;
88 
89  axis_iterator_thread[axis] = it;
90 
91  const size_t input_it = get_id_by_xyz_inkernel(axis_iterator_thread, shape_size, input_shape_offsets);
92 
93  if (it < input_shape[axis])
94  {
95  if constexpr (std::is_same<_DataType_input, std::complex<double>>::value)
96  {
97  const _DataType_input* cmplx_ptr = array_1 + input_it;
98  const double* dbl_ptr = reinterpret_cast<const double*>(cmplx_ptr);
99  in_real = *dbl_ptr;
100  in_imag = *(dbl_ptr + 1);
101  }
102  else
103  {
104  in_real = array_1[input_it];
105  }
106  }
107 
108  const size_t output_local_id = xyz_thread[axis];
109  const double angle = 2.0 * M_PI * it * output_local_id / axis_length;
110 
111  const double angle_cos = cl::sycl::cos(angle);
112  const double angle_sin = cl::sycl::sin(angle);
113 
114  sum_real += in_real * angle_cos + in_imag * angle_sin;
115  sum_imag += -in_real * angle_sin + in_imag * angle_cos;
116  }
117 
118  result[output_id] = _DataType_output(sum_real, sum_imag);
119  };
120 
121  auto kernel_func = [&](cl::sycl::handler& cgh) {
122  cgh.parallel_for<class dpnp_fft_fft_c_kernel<_DataType_input, _DataType_output>>(gws, kernel_parallel_for_func);
123  };
124 
125  event = DPNP_QUEUE.submit(kernel_func);
126 
127 #if 0 // keep this code
128  oneapi::mkl::dft::descriptor<mkl_dft::precision::DOUBLE, mkl_dft::domain::COMPLEX> desc(result_size);
129  desc.set_value(mkl_dft::config_param::FORWARD_SCALE, static_cast<double>(result_size));
130  desc.set_value(mkl_dft::config_param::PLACEMENT, DFTI_NOT_INPLACE); // enum value from MKL C interface
131  desc.commit(DPNP_QUEUE);
132 
133  event = mkl_dft::compute_forward(desc, array_1, result);
134 #endif
135 
136  event.wait();
137 
138  dpnp_memory_free_c(input_shape_offsets);
139  dpnp_memory_free_c(output_shape_offsets);
140  dpnp_memory_free_c(axis_iterator);
141  dpnp_memory_free_c(xyz);
142 
143  return;
144 }
