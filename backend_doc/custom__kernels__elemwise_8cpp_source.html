257 {
258  // parameter `size` used instead `stop` to avoid dependency on array length calculation algorithm
259  // TODO: floating point (and negatives) types from `start` and `step`
260 
261  if (!size)
262  {
263  return;
264  }

275  };

278  cgh.parallel_for<class dpnp_arange_c_kernel<_DataType>>(gws, kernel_parallel_for_func);
